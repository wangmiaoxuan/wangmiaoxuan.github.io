動畫來源

範例程式碼

        

            <div id="animation-container">
                <div id="lottie"></div>
            </div>
            
            <script>
                // 指定 c.json 文件的路径
                const animationUrl = 'c.json'; 

                // 初始化 Lottie 动画
                const animation = lottie.loadAnimation({
                container: document.getElementById('lottie'), // 指定容器
                renderer: 'svg', // 渲染方式
                loop: true, // 是否循环
                autoplay: true, // 是否自动播放
                path: animationUrl // JSON 文件路径
                });
            </script>
