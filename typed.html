            <style>         
                /* 1. 簡單打字效果 */
                .simple-typing {
                    display: inline-block;
                    overflow: hidden;
                    white-space: nowrap;
                    border-right: 2px solid;
                    animation: typing 6s steps(20, end) infinite, blink 0.5s step-end infinite;
                    font-size: 1.5rem;
                }
                @keyframes typing {
                    from { width: 0; }
                    to { width: 100%; }
                }
                @keyframes blink {
                    from, to { border-color: transparent; }
                    50% { border-color: black; }
                }
                /* 2. 漸變字母顯示效果 */
                .fade-typing {
                    display: inline-block;
                    white-space: nowrap;
                    font-size: 1.5rem;
                    animation: fadeIn 6s ease-out forwards infinite;
                }
                @keyframes fadeIn {
                    0% { opacity: 0; }
                    100% { opacity: 1; }
                }
                /* 3. 波浪式顯示效果 */
                .wave-text {
                    display: inline-block;
                    font-size: 1.5rem;
                    white-space: nowrap;
                }
                .wave-text span {
                    display: inline-block;
                    opacity: 0;
                    animation: wave 6s ease-in-out forwards infinite;
                }
                .wave-text span:nth-child(odd) {
                    animation-delay: 0.1s;
                }
                .wave-text span:nth-child(even) {
                    animation-delay: 0.2s;
                }
                @keyframes wave {
                    from { opacity: 0; transform: translateY(20px); }
                    to { opacity: 1; transform: translateY(0); }
                }
                /* 4. 打字並刪除效果 */
                .type-delete {
                    font-size: 1.5rem;
                    overflow: hidden;
                    white-space: nowrap;
                    border-right: 3px solid;
                    animation: typingDelete 6s steps(20) infinite;
                }
                @keyframes typingDelete {
                    0%, 40% { width: 100%; }
                    60%, 100% { width: 0; }
                }
                /* 5. 文字打字加上顏色變化 */
                .color-typing {
                    display: inline-block;
                    overflow: hidden;
                    white-space: nowrap;
                    border-right: 2px solid;
                    font-size: 1.5rem;
                    animation: typingColor 6s steps(20, end) forwards infinite, blink 0.5s step-end infinite;
                }
                @keyframes typingColor {
                    from { width: 0; color: black; }
                    to { width: 100%; color: red; }
                }
            </style>
